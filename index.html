<script>

    (function() {

        // --- Chave utilizada no localStorage para armazenar a contagem

        const STORAGE_KEY = "simpleAccessCounter";

        

        // --- Elementos do DOM

        const counterElement = document.getElementById("visitCount");

        const resetBtn = document.getElementById("resetCounterBtn");

        const refreshBtn = document.getElementById("refreshInfoBtn");

        

        // --- Função para obter o valor atual salvo (sempre retorna um número inteiro >= 0)

        function getCurrentCount() {

            const stored = localStorage.getItem(STORAGE_KEY);

            if (stored === null || isNaN(parseInt(stored, 10))) {

                return 0;

            }

            return parseInt(stored, 10);

        }

        

        // --- Função para salvar um novo valor no localStorage e atualizar o display

        function setCountAndDisplay(newCount) {

            // Garante que o valor é um número inteiro não negativo

            const safeCount = Math.max(0, Math.floor(newCount));

            localStorage.setItem(STORAGE_KEY, safeCount);

            // Atualiza o elemento visual

            if (counterElement) {

                counterElement.textContent = safeCount;

            }

            return safeCount;

        }

        

        // --- Incrementa o contador em +1 (ação principal do acesso à página)

        function incrementCounter() {

            let current = getCurrentCount();

            // incrementa de forma segura

            const newValue = current + 1;

            setCountAndDisplay(newValue);

            return newValue;

        }

        

        // --- Reseta o contador para 0 (apenas a pedido do usuário)

        function resetCounter() {

            // Confirmação amigável para evitar reset acidental

            const confirmReset = confirm("⚠️ Tem certeza que deseja reiniciar o contador de acessos para 0?\n\nO próximo carregamento da página irá somar +1 normalmente.");

            if (confirmReset) {

                setCountAndDisplay(0);

                // Mensagem visual opcional (alerta suave)

                alert("✅ Contador reiniciado para 0. Atualize a página ou clique em 'Atualizar contador' para ver o próximo acesso.");

            }

        }

        

        // --- Atualiza o contador manualmente (sem incremento automático, apenas exibe o valor atual)

        function refreshDisplayOnly() {

            const currentVal = getCurrentCount();

            setCountAndDisplay(currentVal); // apenas re-exibe o valor salvo

            // Pequeno feedback sutil

            if (counterElement) {

                counterElement.style.transform = "scale(1.05)";

                setTimeout(() => {

                    if (counterElement) counterElement.style.transform = "";

                }, 200);

            }

        }

        

        // --- Lógica principal: Quando a página carrega, devemos incrementar o contador (contabilizar acesso)

        // Porém precisamos garantir que isso aconteça apenas uma vez por carregamento.

        // Usamos o evento 'DOMContentLoaded' para garantir que o DOM já está pronto, mas a execução

        // do script já está no final do body então o incremento pode ser feito imediatamente.

        // Para garantir que não haja incremento duplicado (caso scripts sejam chamados novamente) 

        // usamos uma flag ou executamos apenas uma vez.

        // Abaixo, realizamos o incremento e atualização imediatamente.

        

        // Executa o incremento referente a este acesso/recarregamento

        function handlePageVisit() {

            let currentBefore = getCurrentCount();

            // Incrementa contador

            const newCount = incrementCounter();

            console.log(`[Contador] Visita registrada: ${currentBefore} → ${newCount}`);

            // Efeito visual extra no display

            if (counterElement) {

                counterElement.style.animation = "none";

                counterElement.offsetHeight; // força reflow

                counterElement.style.transition = "0.15s ease-in-out";

                counterElement.style.textShadow = "0 0 12px #ffb347";

                setTimeout(() => {

                    if (counterElement) counterElement.style.textShadow = "";

                }, 300);

            }

        }

        

        // Adiciona efeito CSS dinâmico (animação opcional)

        const styleSheet = document.createElement("style");

        styleSheet.textContent = `

            .counter-number {

                transition: all 0.2s ease;

            }

        `;

        document.head.appendChild(styleSheet);

        

        // Verifica se essa execução é a primeira para a sessão de carregamento

        // Como o script roda uma única vez, podemos chamar handlePageVisit diretamente.

        // Mas precisamos tomar cuidado: em alguns navegadores, se a página for carregada via BFCache (back-forward cache)

        // o script pode não executar novamente? Nesse caso o contador não seria incrementado ao voltar da página.

        // Para garantir robustez, podemos usar o evento 'pageshow' que cobre BFCache, mas isso pode causar duplicidade?

        // Para um contador simples, o incremento no carregamento inicial cobre 99% dos casos.

        // A fim de cobrir também quando o usuário volta pela navegação (bfcache) sem recarregar, usaremos o pageshow

        // com flag para evitar incremento duplicado.

        

        let hasIncremented = false;

        

        function registerVisit() {

            if (!hasIncremented) {

                hasIncremented = true;

                handlePageVisit();

            }

        }

        

        // Padrão: executa no carregamento normal

        if (document.readyState === "loading") {

            document.addEventListener("DOMContentLoaded", registerVisit);

        } else {

            registerVisit();

        }

        

        // Para cobrir cenários de "back/forward cache" (bfcache) onde o DOMContentLoaded não dispara novamente,

        // mas a página é restaurada do cache sem executar scripts novamente, mas o evento pageshow funciona.

        // Precisamos evitar duplicidade caso o incremento já tenha ocorrido.

        window.addEventListener("pageshow", function(event) {

            // Se a página foi carregada do cache (bfcache) e o incremento ainda não foi feito nessa restauração,

            // ou se é uma navegação normal, mas o evento pode disparar depois do DOMContentLoaded.

            // Utilizamos a flag para garantir incremento apenas uma vez por "exibição real".

            if (!hasIncremented) {

                hasIncremented = true;

                handlePageVisit();

            } else {

                // Caso já tenha incrementado (ex: DOMContentLoaded já executou), apenas sincroniza display

                // pois algum outro evento pode ter alterado o localStorage em outra aba.

                const currentVal = getCurrentCount();

                if (counterElement && counterElement.textContent != currentVal) {

                    setCountAndDisplay(currentVal);

                }

            }

        });

        

        // --- Event listeners para os botões

        if (resetBtn) {

            resetBtn.addEventListener("click", resetCounter);

        }

        

        if (refreshBtn) {

            refreshBtn.addEventListener("click", function() {

                refreshDisplayOnly();

                // feedback adicional: mostra o valor atual sem incrementar

            });

        }

        

        // Sincronização entre abas/janelas: se o localStorage for alterado em outra aba,

        // atualizamos o display para refletir o valor mais recente, sem incrementar automaticamente.

        window.addEventListener("storage", function(event) {

            if (event.key === STORAGE_KEY) {

                // Outra aba modificou o contador. Atualizamos o display local.

                const newValue = event.newValue !== null ? parseInt(event.newValue, 10) : 0;

                if (!isNaN(newValue) && counterElement) {

                    counterElement.textContent = newValue;

                }

            }

        });

        

        // Pequena correção: Se por acaso o display não foi atualizado por algum motivo no load,

        // garantimos que ele mostre o número correto.

        setTimeout(() => {

            const storedValue = getCurrentCount();

            if (counterElement && counterElement.textContent !== storedValue.toString()) {

                counterElement.textContent = storedValue;

            }

        }, 50);

    })();

</script>
